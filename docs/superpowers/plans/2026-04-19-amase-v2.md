```ts
import { readdir, readFile, stat } from "node:fs/promises";
import { join, dirname } from "node:path";
import { fileURLToPath } from "node:url";

const FIXTURES_DIR = join(dirname(fileURLToPath(import.meta.url)), "..", "fixtures");

export interface Fixture {
  id: string;
  prompt: string;
  beforeTree: Map<string, string>;
  expectedPatch: string;
  testsDir: string;
}

export async function listFixtures(): Promise<string[]> {
  const entries = await readdir(FIXTURES_DIR, { withFileTypes: true });
  return entries.filter((e) => e.isDirectory()).map((e) => e.name).sort();
}

async function readTree(root: string, acc = new Map<string, string>(), rel = ""): Promise<Map<string, string>> {
  const entries = await readdir(join(root, rel), { withFileTypes: true });
  for (const e of entries) {
    const relPath = rel ? `${rel}/${e.name}` : e.name;
    if (e.isDirectory()) await readTree(root, acc, relPath);
    else acc.set(relPath, await readFile(join(root, relPath), "utf8"));
  }
  return acc;
}

export async function loadFixture(id: string): Promise<Fixture> {
  const dir = join(FIXTURES_DIR, id);
  await stat(dir); // throws if missing
  const prompt = await readFile(join(dir, "prompt.md"), "utf8");
  const beforeTree = await readTree(join(dir, "before"));
  const expectedPatch = await readFile(join(dir, "expected.patch"), "utf8");
  return { id, prompt, beforeTree, expectedPatch, testsDir: join(dir, "tests") };
}
```

- [ ] **Step 3: Implement `packages/bench/src/adapters/amase.ts`**

Wire to `Orchestrator` from `@amase/core`, capture per-node metrics already emitted, materialize the fixture `before/` tree into a temp worktree, pass `prompt.md` as the request, run resulting workspace against fixture `tests/`, compute `diff_similarity` against `expected.patch`.

```ts
import { mkdtemp, writeFile, mkdir, rm, readFile } from "node:fs/promises";
import { tmpdir } from "node:os";
import { join, dirname } from "node:path";
import { spawnSync } from "node:child_process";
import type { Fixture } from "../fixtures.js";
import type { BenchResult } from "../types.js";
import { Orchestrator } from "@amase/core";
import { buildAgentRegistry } from "@amase/agents";
import { StubLlmClient, AnthropicClient, type LlmClient } from "@amase/llm";
import { DAGStore, DecisionLog } from "@amase/memory";
import { schemaValidator, patchSafetyValidator, typecheckValidator, unitTestsValidator } from "@amase/validators";

export interface RunOpts {
  runId: string;
  live?: boolean;
}

async function materializeTree(dest: string, tree: Map<string, string>): Promise<void> {
  for (const [rel, content] of tree) {
    const abs = join(dest, rel);
    await mkdir(dirname(abs), { recursive: true });
    await writeFile(abs, content);
  }
}

function buildLlm(live: boolean): LlmClient {
  return live ? new AnthropicClient() : new StubLlmClient(async () => JSON.stringify({ taskId: "t", patches: [] }));
}

function diffSimilarity(_workspace: string, _expected: string): number {
  // v1: return 1.0 on pass, 0.0 on fail — improved in Task 4.
  return 1.0;
}

export async function runAmase(fx: Fixture, opts: RunOpts): Promise<BenchResult> {
  const workspace = await mkdtemp(join(tmpdir(), `amase-bench-${fx.id}-`));
  await materializeTree(workspace, fx.beforeTree);

  const llm = buildLlm(opts.live ?? false);
  const store = new DAGStore();
  const log = new DecisionLog(join(workspace, ".amase"));
  const validators = [schemaValidator, patchSafetyValidator, typecheckValidator, unitTestsValidator];
  const orch = new Orchestrator({ llm, store, decisionLog: log, validators, registry: buildAgentRegistry(llm) });

  const start = Date.now();
  let pass = false;
  let tokensIn = 0;
  let tokensOut = 0;
  let retries = 0;
  let error: string | undefined;
  try {
    const planned = await orch.plan({ request: fx.prompt, workspacePath: workspace });
    const runRes = await orch.execute({ dagId: planned.dagId });
    tokensIn = runRes.metrics.tokensIn;
    tokensOut = runRes.metrics.tokensOut;
    retries = runRes.metrics.retries;
    // Run fixture tests against workspace
    const test = spawnSync("pnpm", ["vitest", "run", "--dir", fx.testsDir], { cwd: workspace, encoding: "utf8" });
    pass = test.status === 0;
    if (!pass) error = test.stdout + test.stderr;
  } catch (e) {
    error = (e as Error).message;
  }
  const wallMs = Date.now() - start;
  await rm(workspace, { recursive: true, force: true });

  return {
    runId: opts.runId,
    timestamp: new Date().toISOString(),
    taskId: fx.id,
    stack: "amase",
    pass,
    tokensIn,
    tokensOut,
    wallMs,
    diffSimilarity: pass ? diffSimilarity(workspace, fx.expectedPatch) : 0,
    retries,
    error,
  };
}
```

Note: `Orchestrator.execute` needs to expose aggregate `tokensIn/tokensOut/retries` — verify existing return shape; if missing, widen it in `packages/core/src/orchestrator.ts` as part of this task and add a unit test there.

- [ ] **Step 3: Implement using normalized token Jaccard**

```ts
function tokens(s: string): Set<string> {
  return new Set(s.split(/\s+/).filter(Boolean));
}

export function diffSimilarity(a: string, b: string): number {
  const ta = tokens(a);
  const tb = tokens(b);
  if (ta.size === 0 && tb.size === 0) return 1;
  let inter = 0;
  for (const t of ta) if (tb.has(t)) inter++;
  const union = ta.size + tb.size - inter;
  return union === 0 ? 0 : inter / union;
}
```

Jaccard was chosen because it's deterministic, dependency-free, and robust to reordering. If the team wants per-hunk structural similarity later, swap for a real diff parser — do not do that now (YAGNI).

```ts
import { mkdtemp, rm, readFile, writeFile, mkdir } from "node:fs/promises";
import { tmpdir } from "node:os";
import { join, dirname } from "node:path";
import { spawn } from "node:child_process";
import type { Fixture } from "../fixtures.js";
import type { BenchResult, RunOpts } from "../types.js";
import { diffSimilarity } from "../diff-similarity.js";

async function materialize(dest: string, tree: Map<string, string>): Promise<void> {
  for (const [rel, content] of tree) {
    const abs = join(dest, rel);
    await mkdir(dirname(abs), { recursive: true });
    await writeFile(abs, content);
  }
}

interface TranscriptEvent {
  type: string;
  usage?: { input_tokens: number; output_tokens: number };
}

export async function runSuperpowers(fx: Fixture, opts: RunOpts): Promise<BenchResult> {
  const workspace = await mkdtemp(join(tmpdir(), `sp-bench-${fx.id}-`));
  await materialize(workspace, fx.beforeTree);
  const transcriptPath = join(workspace, "transcript.jsonl");

  const start = Date.now();
  const args = [
    "--print",
    "--output-format=stream-json",
    "--permission-mode=acceptAll",
    `--transcript-out=${transcriptPath}`,
    "--", fx.prompt,
  ];
  const child = spawn("claude", args, { cwd: workspace, env: { ...process.env, CLAUDE_PLUGINS: "superpowers" } });
  let pass = false;
  let error: string | undefined;
  await new Promise<void>((resolve) => {
    child.on("close", () => resolve());
    child.on("error", (e) => { error = e.message; resolve(); });
  });

  let tokensIn = 0;
  let tokensOut = 0;
  try {
    const raw = await readFile(transcriptPath, "utf8");
    for (const line of raw.split("\n")) {
      if (!line.trim()) continue;
      const ev = JSON.parse(line) as TranscriptEvent;
      if (ev.usage) {
        tokensIn += ev.usage.input_tokens;
        tokensOut += ev.usage.output_tokens;
      }
    }
  } catch (e) { error ??= (e as Error).message; }

  // Run fixture tests
```

```ts
import type { BenchResult } from "./types.js";

export type Headline =
  | { status: "insufficient_signal"; bothPassed: number }
  | { status: "regression"; passRateDelta: number }
  | { status: "ok"; tokenDelta: number; timeDelta: number; passRateDelta: number; bothPassed: number };

export function reportHeadline(results: BenchResult[]): Headline {
  const byTask = new Map<string, { amase?: BenchResult; superpowers?: BenchResult }>();
  for (const r of results) {
    const entry = byTask.get(r.taskId) ?? {};
    entry[r.stack] = r;
    byTask.set(r.taskId, entry);
  }
  let amasePassed = 0;
  let spPassed = 0;
  let tokenA = 0, tokenS = 0, timeA = 0, timeS = 0;
  let bothPassed = 0;
  const total = byTask.size;
  for (const [, { amase, superpowers }] of byTask) {
    if (amase?.pass) amasePassed++;
    if (superpowers?.pass) spPassed++;
    if (amase?.pass && superpowers?.pass) {
      bothPassed++;
      tokenA += amase.tokensIn + amase.tokensOut;
      tokenS += superpowers.tokensIn + superpowers.tokensOut;
      timeA += amase.wallMs;
      timeS += superpowers.wallMs;
    }
  }
  const passRateDelta = total === 0 ? 0 : (amasePassed - spPassed) / total;
  if (passRateDelta < 0) return { status: "regression", passRateDelta };
  if (bothPassed < 5) return { status: "insufficient_signal", bothPassed };
  return {
    status: "ok",
    tokenDelta: tokenS === 0 ? 0 : (tokenS - tokenA) / tokenS,
    timeDelta: timeS === 0 ? 0 : (timeS - timeA) / timeS,
    passRateDelta,
    bothPassed,
  };
}
```

```ts
// packages/ambiguity/src/rubric.ts
import type { DecisionDraft, RubricResult } from "./types.js";

export interface RubricConfig {
  fileCountThreshold?: number; // default 3
  extraPredicates?: Array<(d: DecisionDraft) => string | null>; // CLAUDE.md-driven
}

export function scoreDecision(d: DecisionDraft, cfg: RubricConfig = {}): RubricResult {
  const reasons: string[] = [];
  const threshold = cfg.fileCountThreshold ?? 3;
  if (d.changesPublicApi) reasons.push("public API");
  if (d.changesDataModel) reasons.push("data model");
  if (d.addsDependency) reasons.push(`new dep: ${d.addsDependency}`);
  if (d.crossesModuleBoundary) reasons.push("module boundary");
  if (d.fileCount > threshold) reasons.push(`fileCount=${d.fileCount}`);
  if (d.crossCuttingConcern !== "none") reasons.push(`cross-cutting: ${d.crossCuttingConcern}`);
  for (const p of cfg.extraPredicates ?? []) {
    const r = p(d);
    if (r) reasons.push(r);
  }
  const score = reasons.length;
  const decision: RubricResult["decision"] = score >= 2 ? "ask" : score === 0 ? "decide" : "tier2";
  return { score, reasons, decision };
}
```

```ts
    const questions: UserQuestion[] = [];
    const resolved: DecisionDraft[] = [];
    for (const d of decisions) {
      const r = scoreDecision(d);
      if (r.decision === "decide") { resolved.push(d); continue; }
      if (r.decision === "tier2") {
        const t2 = await classifyDecision(this.llm, d);
        if (t2 === "decide") { resolved.push(d); continue; }
      }
      questions.push(this.toQuestion(d));
    }
    return { resolved, questions };
  }

  private toQuestion(d: DecisionDraft): UserQuestion {
    // Architect itself proposes 3 options; reuse stored draft options if provided,
    // otherwise request a fresh LLM call to produce them. Default generator:
    return {
      questionId: crypto.randomUUID(),
      runId: "",
      question: d.summary,
      options: [
        { label: "Option A", detail: "…" },
        { label: "Option B", detail: "…" },
        { label: "Option C", detail: "…" },
      ],
      recommended: 0,
      reason: d.summary,
    };
  }
}
```

```ts
// packages/memory/src/decision-query.ts
import type { DecisionDraft } from "@amase/ambiguity";
import { dirname, extname } from "node:path";

export function touchedPathsSignature(d: DecisionDraft): string[] {
  const buckets = new Map<string, string>();
  for (const p of d.touchedPaths) {
    const key = `${dirname(p)}::${extname(p)}`;
    buckets.set(key, `${dirname(p)}/*${extname(p)}`);
  }
  return Array.from(buckets.values()).sort();
}

export interface LoggedDecision {
  id: string;
  kind: DecisionDraft["kind"];
  signature: string[];
  answer: { choice: 0 | 1 | 2 };
}

export async function findReusableDecision(
  log: LoggedDecision[],
  d: DecisionDraft,
): Promise<LoggedDecision | null> {
  const sig = touchedPathsSignature(d);
  for (const entry of log) {
    if (entry.kind !== d.kind) continue;
    if (entry.signature.length !== sig.length) continue;
    if (entry.signature.every((s, i) => s === sig[i])) return entry;
  }
  return null;
}
```
